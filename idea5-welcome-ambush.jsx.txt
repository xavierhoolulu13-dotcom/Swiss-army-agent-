import { useState, useEffect, useRef } from "react";

const PIPELINE = [
  { id: "whmcs", icon: "⚡", label: "WHMCS TRIGGER", sub: "New client provisioned", color: "#ffd700", duration: 800 },
  { id: "provision", icon: "🐳", label: "DOCKER PROVISION", sub: "44-node deploy sequence", color: "#00c8ff", duration: 1200 },
  { id: "amanda", icon: "🧠", label: "AMANDA SCRIPT GEN", sub: "Personalised video script", color: "#00ff88", duration: 900 },
  { id: "heygen", icon: "🎬", label: "HEYGEN RENDER", sub: "Avatar video generated", color: "#ff6b00", duration: 1400 },
  { id: "telegram", icon: "📩", label: "TELEGRAM DELIVER", sub: "Before they log in", color: "#2aabee", duration: 700 },
];

const NODE_COUNT = 44;

export default function WelcomeAmbush() {
  const [name, setName] = useState("");
  const [company, setCompany] = useState("");
  const [plan, setPlan] = useState("MICRO·SIM");
  const [running, setRunning] = useState(false);
  const [activeStep, setActiveStep] = useState(-1);
  const [completed, setCompleted] = useState([]);
  const [nodesFired, setNodesFired] = useState(0);
  const [preview, setPreview] = useState(null);
  const [done, setDone] = useState(false);
  const nodesRef = useRef(null);

  const run = async () => {
    if (!name || !company) return;
    setRunning(true);
    setCompleted([]);
    setActiveStep(-1);
    setNodesFired(0);
    setPreview(null);
    setDone(false);

    for (let i = 0; i < PIPELINE.length; i++) {
      setActiveStep(i);

      if (i === 1) {
        // Animate node counter
        let n = 0;
        const interval = setInterval(() => {
          n += Math.ceil(Math.random() * 4);
          if (n >= NODE_COUNT) { n = NODE_COUNT; clearInterval(interval); }
          setNodesFired(n);
        }, PIPELINE[i].duration / NODE_COUNT * 2);
      }

      if (i === 2) {
        setPreview(`Hey ${name}! Welcome to SimAI.\n\nYour ${plan} plan is live on our infrastructure. Your Amanda AI orchestrator is initialised, your Qdrant memory is seeded, and your first n8n workflows are already running.\n\nThis is your personalised onboarding from your AI team — no human involved.\n\nLet's build something insane. 🔥\n— Amanda`);
      }

      await new Promise(r => setTimeout(r, PIPELINE[i].duration));
      setCompleted(c => [...c, i]);
    }

    setActiveStep(-1);
    setDone(true);
    setRunning(false);
  };

  const css = `
    @import url('https://fonts.googleapis.com/css2?family=Syne:wght@400;700;800&family=DM+Mono:wght@300;400;500&display=swap');
    * { box-sizing: border-box; margin: 0; padding: 0; }
    .wa { font-family: 'DM Mono', monospace; background: #08060e; min-height: 100vh; color: #e2dff0; }
    @keyframes pulse { 0%,100%{opacity:1;transform:scale(1)} 50%{opacity:0.6;transform:scale(0.94)} }
    @keyframes slideIn { from{opacity:0;transform:translateX(-12px)} to{opacity:1;transform:translateX(0)} }
    @keyframes fadeUp { from{opacity:0;transform:translateY(10px)} to{opacity:1;transform:translateY(0)} }
    @keyframes nodeFlash { 0%{opacity:0.3} 50%{opacity:1} 100%{opacity:0.3} }
    @keyframes glowPulse { 0%,100%{box-shadow:0 0 12px rgba(255,215,0,0.2)} 50%{box-shadow:0 0 28px rgba(255,215,0,0.5)} }
    .run-btn { cursor: pointer; border: none; transition: all 0.2s; }
    .run-btn:hover:not(:disabled) { transform: scale(1.02); }
    .run-btn:disabled { opacity: 0.4; cursor: not-allowed; }
    input, select { outline: none; }
  `;

  return (
    <>
      <style>{css}</style>
      <div className="wa">

        {/* Header */}
        <div style={{
          background: "linear-gradient(135deg, #0d0a1a 0%, #0a0816 100%)",
          borderBottom: "1px solid rgba(255,215,0,0.15)",
          padding: "32px 32px 28px",
        }}>
          <div style={{ maxWidth: "860px", margin: "0 auto" }}>
            <div style={{ fontSize: "10px", letterSpacing: "0.22em", color: "rgba(226,223,240,0.35)", marginBottom: "10px" }}>
              XOS-V3 / OPERATIONS / WELCOME-AMBUSH
            </div>
            <h1 style={{
              fontFamily: "'Syne', sans-serif",
              fontSize: "clamp(32px, 6vw, 58px)",
              fontWeight: 800,
              lineHeight: 0.95,
              marginBottom: "10px",
              color: "#fff",
            }}>
              WHMCS → HEYGEN<br />
              <span style={{ color: "#ffd700" }}>WELCOME AMBUSH</span>
            </h1>
            <div style={{ fontSize: "12px", color: "rgba(226,223,240,0.45)", letterSpacing: "0.06em" }}>
              Client provisions → AI video in their Telegram before they log in
            </div>
          </div>
        </div>

        <div style={{ maxWidth: "860px", margin: "0 auto", padding: "36px 32px" }}>

          {/* Input form */}
          <div style={{
            display: "grid",
            gridTemplateColumns: "1fr 1fr auto",
            gap: "12px",
            marginBottom: "40px",
            alignItems: "end",
          }}>
            {[
              { label: "CLIENT NAME", val: name, set: setName, ph: "Alex Rivera" },
              { label: "COMPANY", val: company, set: setCompany, ph: "Apex Labs" },
            ].map(({ label, val, set, ph }) => (
              <div key={label}>
                <div style={{ fontSize: "9px", letterSpacing: "0.2em", color: "rgba(226,223,240,0.35)", marginBottom: "6px" }}>{label}</div>
                <input
                  value={val}
                  onChange={e => set(e.target.value)}
                  placeholder={ph}
                  style={{
                    width: "100%",
                    background: "#0f0d1a",
                    border: "1px solid rgba(255,215,0,0.2)",
                    borderRadius: "4px",
                    padding: "10px 14px",
                    color: "#ffd700",
                    fontFamily: "'DM Mono', monospace",
                    fontSize: "13px",
                  }}
                />
              </div>
            ))}
            <button
              className="run-btn"
              onClick={run}
              disabled={running || !name || !company}
              style={{
                padding: "10px 24px",
                background: running ? "rgba(255,215,0,0.1)" : "#ffd700",
                color: running ? "#ffd700" : "#08060e",
                border: "1px solid #ffd700",
                borderRadius: "4px",
                fontFamily: "'DM Mono', monospace",
                fontSize: "12px",
                letterSpacing: "0.12em",
                fontWeight: 500,
                animation: running ? "glowPulse 1.4s infinite" : "none",
              }}>
              {running ? "FIRING…" : "▶ TRIGGER"}
            </button>
          </div>

          {/* Pipeline */}
          <div style={{ display: "grid", gap: "12px", marginBottom: "36px" }}>
            {PIPELINE.map((step, i) => {
              const isActive = activeStep === i;
              const isDone = completed.includes(i);
              return (
                <div
                  key={step.id}
                  style={{
                    display: "flex",
                    alignItems: "center",
                    gap: "16px",
                    padding: "16px 20px",
                    background: isDone ? `rgba(${step.color === "#ffd700" ? "255,215,0" : step.color === "#00c8ff" ? "0,200,255" : step.color === "#00ff88" ? "0,255,136" : step.color === "#ff6b00" ? "255,107,0" : "42,171,238"},0.06)` : isActive ? "#0f0d1a" : "#0a0815",
                    border: `1px solid ${isDone ? step.color : isActive ? `${step.color}44` : "rgba(226,223,240,0.06)"}`,
                    borderRadius: "6px",
                    transition: "all 0.3s",
                    animation: isDone ? "slideIn 0.35s ease" : "none",
                  }}>

                  {/* Icon */}
                  <div style={{
                    width: "40px",
                    height: "40px",
                    borderRadius: "8px",
                    background: isDone ? step.color : isActive ? `${step.color}22` : "rgba(226,223,240,0.05)",
                    display: "flex",
                    alignItems: "center",
                    justifyContent: "center",
                    fontSize: "18px",
                    transition: "all 0.3s",
                    animation: isActive ? "pulse 0.8s infinite" : "none",
                    flexShrink: 0,
                  }}>
                    {step.icon}
                  </div>

                  {/* Label */}
                  <div style={{ flex: 1 }}>
                    <div style={{
                      fontFamily: "'Syne', sans-serif",
                      fontWeight: 700,
                      fontSize: "14px",
                      letterSpacing: "0.08em",
                      color: isDone ? step.color : isActive ? step.color : "rgba(226,223,240,0.4)",
                      transition: "color 0.3s",
                    }}>
                      {step.label}
                      {step.id === "provision" && isActive && (
                        <span style={{ marginLeft: "10px", fontSize: "11px", fontWeight: 400 }}>
                          {nodesFired}/{NODE_COUNT} nodes
                        </span>
                      )}
                    </div>
                    <div style={{ fontSize: "11px", color: "rgba(226,223,240,0.3)", marginTop: "2px" }}>{step.sub}</div>
                  </div>

                  {/* Status */}
                  <div style={{ flexShrink: 0 }}>
                    {isDone && <span style={{ color: step.color, fontSize: "18px" }}>✓</span>}
                    {isActive && (
                      <div style={{ display: "flex", gap: "3px", alignItems: "center" }}>
                        {[0, 0.15, 0.3].map(d => (
                          <div key={d} style={{
                            width: "5px", height: "5px", borderRadius: "50%",
                            background: step.color,
                            animation: `pulse 0.9s ${d}s infinite`,
                          }} />
                        ))}
                      </div>
                    )}
                    {!isDone && !isActive && <span style={{ color: "rgba(226,223,240,0.12)", fontSize: "18px" }}>○</span>}
                  </div>
                </div>
              );
            })}
          </div>

          {/* Script preview */}
          {preview && (
            <div style={{
              animation: "fadeUp 0.5s ease",
              background: "#0d1a14",
              border: "1px solid #00ff88",
              borderRadius: "8px",
              overflow: "hidden",
            }}>
              <div style={{
                padding: "12px 18px",
                background: "rgba(0,255,136,0.08)",
                borderBottom: "1px solid rgba(0,255,136,0.15)",
                display: "flex",
                alignItems: "center",
                justifyContent: "space-between",
              }}>
                <div>
                  <div style={{ fontFamily: "'Syne', sans-serif", fontWeight: 700, fontSize: "13px", color: "#00ff88", letterSpacing: "0.1em" }}>
                    📩 TELEGRAM PREVIEW — @{name.toLowerCase().replace(/\s/g, "_")}
                  </div>
                  <div style={{ fontSize: "10px", color: "rgba(226,223,240,0.35)", marginTop: "2px" }}>Delivered before first login</div>
                </div>
                {done && <div style={{ fontSize: "10px", color: "#00ff88", letterSpacing: "0.1em", display: "flex", alignItems: "center", gap: "6px" }}>
                  <span style={{ width: "6px", height: "6px", borderRadius: "50%", background: "#00ff88", display: "inline-block" }} />
                  DELIVERED
                </div>}
              </div>
              <div style={{
                padding: "20px 24px",
                fontSize: "13px",
                lineHeight: "1.7",
                color: "rgba(226,223,240,0.8)",
                whiteSpace: "pre-wrap",
              }}>
                {preview}
              </div>
              {done && (
                <div style={{
                  padding: "12px 24px",
                  borderTop: "1px solid rgba(0,255,136,0.1)",
                  fontSize: "10px",
                  color: "rgba(226,223,240,0.25)",
                  letterSpacing: "0.1em",
                  display: "flex",
                  justifyContent: "space-between",
                }}>
                  <span>0 HUMANS INVOLVED</span>
                  <span>AMANDA ORCHESTRATION KERNEL</span>
                  <span>XOS-V3 / WELCOME-AMBUSH</span>
                </div>
              )}
            </div>
          )}

        </div>
      </div>
    </>
  );
}
