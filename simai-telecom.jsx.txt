const WaveCanvas = () => {
  const ref = useRef(null);
  useEffect(() => {
    const canvas = ref.current;
    const ctx = canvas.getContext("2d");
    let frame = 0;
    const waves = [
      { amp: 18, freq: 0.018, speed: 0.022, color: "rgba(0,200,255,0.13)", y: 0.45 },
      { amp: 12, freq: 0.025, speed: 0.031, color: "rgba(0,255,136,0.09)", y: 0.55 },
      { amp: 22, freq: 0.012, speed: 0.016, color: "rgba(255,107,0,0.07)", y: 0.5 },
    ];
    let raf;
    const draw = () => {
      canvas.width = canvas.offsetWidth;
      canvas.height = canvas.offsetHeight;
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      waves.forEach((w) => {
        ctx.beginPath();
        ctx.strokeStyle = w.color;
        ctx.lineWidth = 1.5;
        for (let x = 0; x <= canvas.width; x += 2) {
          const y = canvas.height * w.y + Math.sin(x * w.freq + frame * w.speed) * w.amp;
          x === 0 ? ctx.moveTo(x, y) : ctx.lineTo(x, y);
        }
        ctx.stroke();
      });
      frame++;
      raf = requestAnimationFrame(draw);
    };
    draw();
    return () => cancelAnimationFrame(raf);
  }, []);
  return (
    <canvas
      ref={ref}
      style={{ position: "absolute", inset: 0, width: "100%", height: "100%", pointerEvents: "none" }}
    />
  );
};
