const Sparkline = ({ data, color }) => {
  const w = 80, h = 28;
  const min = Math.min(...data), max = Math.max(...data), range = max - min || 1;
  const pts = data.map((v, i) => `${(i / (data.length - 1)) * w},${h - ((v - min) / range) * h}`).join(" ");
  return (
    <svg width={w} height={h} style={{ overflow: "visible" }}>
      <polyline points={pts} fill="none" stroke={color} strokeWidth="1.5" strokeLinejoin="round" />
    </svg>
  );
};

export default function ContentMarket() {

    ...t,
    history: Array.from({ length: 20 }, (_, i) => t.price * (0.88 + Math.random() * 0.24)),
    change: ((t.price - t.prev) / t.prev * 100),

  const eventRef = useRef([]);
  const tickerRef = useRef(tickers);

    const interval = setInterval(() => {

        return {
          ...t,
          prev: t.price,
          price: +newPrice.toFixed(2),
          change: ((newPrice - t.prev) / t.prev * 100),
          volume: t.volume + Math.floor(Math.random() * 500),
          history: [...t.history.slice(-19), newPrice],
        };

      setRsisTick(r => r + 1);

      setTickers(prev => prev.map(t =>
        t.id === e.ticker ? { ...t, price: +(t.price + e.impact).toFixed(2), change: e.impact } : t
      ));
    }, 4500);

    const total = tickers.reduce((s, t) => s + t.price * (portfolio[t.id] || 0), 0);
    setMarketCap(total);
  }, [tickers, portfolio]);

    @import url('https://fonts.googleapis.com/css2?family=IBM+Plex+Mono:wght@300;400;500;600&family=Bebas+Neue&display=swap');

    @keyframes flash-up { 0%{background:rgba(0,255,136,0.15)} 100%{background:transparent} }
    @keyframes flash-down { 0%{background:rgba(255,65,65,0.15)} 100%{background:transparent} }

  `;

      <style>{css}</style>
      <div className="cm">

        {/* Top bar */}

          background: "#060a08",
          borderBottom: "1px solid rgba(200,216,200,0.08)",

          display: "flex",
          alignItems: "center",

          </div>
          <div style={{ display: "flex", gap: "24px" }}>
            {[
              ["MARKET CAP", `$${(totalMarketCap / 1000).toFixed(1)}K`],
              ["ACTIVE TICKERS", tickers.length],
              ["TICK", `#${rsisTick}`],
              ["PORTFOLIO", `$${marketCap.toFixed(0)}`],
            ].map(([k, v]) => (
              <div key={k} style={{ display: "flex", gap: "6px" }}>
                <span style={{ color: "rgba(200,216,200,0.3)" }}>{k}:</span>
                <span style={{ color: "#00ff88" }}>{v}</span>
              </div>
            ))}
          </div>
        </div>

        {/* Ticker tape */}
        <div style={{
          background: "#030504",
          borderBottom: "1px solid rgba(0,255,136,0.1)",
          padding: "6px 0",
          overflow: "hidden",
          whiteSpace: "nowrap",
        }}>
          <div style={{
            display: "inline-flex",
            gap: "32px",
            animation: "none",
            fontSize: "10px",
            letterSpacing: "0.08em",
            padding: "0 20px",
          }}>
            {[...tickers, ...tickers].map((t, i) => (
              <span key={i}>
                <span style={{ color: "rgba(200,216,200,0.5)" }}>{t.id}</span>
                {" "}
                <span style={{ color: t.change >= 0 ? "#00ff88" : "#ff4141" }}>
                  ${t.price.toFixed(2)} {t.change >= 0 ? "▲" : "▼"}{Math.abs(t.change).toFixed(2)}%
                </span>
              </span>
            ))}
          </div>
        </div>

        <div style={{ display: "grid", gridTemplateColumns: "1fr 320px", gap: "0", maxHeight: "calc(100vh - 88px)" }}>

          {/* Main table */}
          <div style={{ overflow: "auto", borderRight: "1px solid rgba(200,216,200,0.07)" }}>

              display: "grid",

              }}>
                <div style={{ fontSize: "9px", letterSpacing: "0.16em", color: "rgba(200,216,200,0.3)", marginBottom: "8px" }}>
                  SELECTED · {sel.id}

                  {sel.name}

                  letterSpacing: "0.04em",
                  lineHeight: 1,

                }}>
                  ${sel.price.toFixed(2)}
                </div>

                  {sel.change >= 0 ? "▲" : "▼"} {Math.abs(sel.change).toFixed(2)}% this tick

                }}>
                  {[["VOLUME", `${(sel.volume / 1000).toFixed(1)}K`], ["MRR IMPACT", sel.mrr_impact], ["TYPE", sel.type.toUpperCase()], ["IN PORTFOLIO", (portfolio[sel.id] || 0) + " units"]].map(([k, v]) => (
                    <div key={k} style={{
                      padding: "8px",
                      background: "rgba(200,216,200,0.03)",

                      borderRadius: "3px",

                      <div style={{ color: "rgba(200,216,200,0.3)", letterSpacing: "0.1em", fontSize: "9px", marginBottom: "4px" }}>{k}</div>

            ) : (
              <div style={{
                padding: "16px",
                borderBottom: "1px solid rgba(200,216,200,0.07)",
                color: "rgba(200,216,200,0.2)",
                fontSize: "11px",
                flex: "0 0 auto",
              }}>
                Click a ticker to inspect
              </div>
            )}

            {/* RSIS808 Event feed */}
            <div style={{ flex: 1, overflow: "auto", padding: "12px" }}>
              <div style={{ fontSize: "9px", letterSpacing: "0.16em", color: "rgba(200,216,200,0.3)", marginBottom: "10px" }}>
                RSIS808 EVENT STREAM
              </div>
              <div style={{ display: "grid", gap: "6px" }}>
                {events.map(e => (
                  <div
                    key={e.id}
                    style={{
                      padding: "9px 10px",
                      background: `${e.color}08`,
                      border: `1px solid ${e.color}22`,
                      borderRadius: "3px",
                      animation: "slide-in 0.3s ease",
                      fontSize: "10px",
                      lineHeight: 1.5,
                    }}>
                    <div style={{ display: "flex", justifyContent: "space-between", marginBottom: "3px" }}>
                      <span style={{ color: e.color, letterSpacing: "0.08em" }}>{e.type}</span>
                      <span style={{ color: "rgba(200,216,200,0.25)", fontSize: "9px" }}>{e.time}</span>
                    </div>
                    <div style={{ color: "rgba(200,216,200,0.5)" }}>
                      <span style={{ color: "#00ff88", marginRight: "6px" }}>{e.ticker}</span>
                      {e.msg}
                    </div>
                    <div style={{ color: e.impact >= 0 ? "#00ff88" : "#ff4141", marginTop: "3px", fontSize: "9px" }}>
                      PRICE IMPACT: {e.impact >= 0 ? "+" : ""}{e.impact.toFixed(1)}
                    </div>
                  </div>
                ))}
                {events.length === 0 && (
                  <div style={{ color: "rgba(200,216,200,0.2)", fontSize: "10px" }}>Awaiting events…</div>
                )}
              </div>
            </div>
          </div>
        </div>
      </div>
    </>
  );
}
